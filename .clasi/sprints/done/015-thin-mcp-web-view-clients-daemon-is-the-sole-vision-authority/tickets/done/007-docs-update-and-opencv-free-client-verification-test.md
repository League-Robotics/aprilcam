## Implementation Plan

### Verification test (`tests/test_015_opencv_free_clients.py`)

```python
"""Regression guard: mcp_server, web_server, view_cli must not import cv2."""
import builtins
import importlib
import sys
import pytest


_real_import = builtins.__import__


def _block_cv2(name, *args, **kwargs):
    if name == "cv2" or name.startswith("cv2."):
        raise ImportError(f"cv2 import blocked by test: {name!r}")
    return _real_import(name, *args, **kwargs)


@pytest.fixture
def block_cv2_import(monkeypatch):
    # Remove cv2 from sys.modules so cached import doesn't bypass the block.
    sys.modules.pop("cv2", None)
    monkeypatch.setattr(builtins, "__import__", _block_cv2)
    yield
    sys.modules.pop("cv2", None)


def _reload_module(name: str) -> None:
    """Remove from sys.modules and reimport under the cv2 block."""
    sys.modules.pop(name, None)
    importlib.import_module(name)


@pytest.mark.parametrize("module", [
    "aprilcam.server.mcp_server",
    "aprilcam.server.web_server",
    "aprilcam.cli.view_cli",
])
def test_client_module_imports_without_cv2(block_cv2_import, module):
    """Each client module must import cleanly even when cv2 is unavailable."""
    _reload_module(module)   # raises if cv2 is imported anywhere in the module
```

Note: Some modules have deep import trees. If `mcp_server` imports a module that
transitively imports cv2 (e.g., `aprilcam.core.detection` may import cv2), those
transitive imports must also be fixed before this test will pass. The implementer
must run the test after each of tickets 002-006 to catch transitive issues early.
